import numpy as np
import pandas as pd

def calculate_macd(data, fast_period=12, slow_period=26, signal_period=9):
    """
    Calculate MACD indicator for given price data.
    
    Args:
        data: Array or list of closing prices
        fast_period: Fast EMA period (default 12)
        slow_period: Slow EMA period (default 26)
        signal_period: Signal EMA period (default 9)
    
    Returns:
        dict: MACD line, signal line, and histogram
    """
    if not data or len(data) < slow_period:
        return None
    
    # Convert to numpy array for vectorized operations
    prices = np.array(data)
    
    # Calculate EMAs using pandas
    df = pd.DataFrame({'close': prices})
    
    # Calculate exponential moving averages
    ema_fast = df['close'].ewm(span=fast_period, adjust=False).mean()
    ema_slow = df['close'].ewm(span=slow_period, adjust=False).mean()
    
    # Calculate MACD line
    macd_line = ema_fast - ema_slow
    
    # Calculate signal line
    signal_line = macd_line.ewm(span=signal_period, adjust=False).mean()
    
    # Calculate histogram
    histogram = macd_line - signal_line
    
    return {
        'macd_line': macd_line.tolist(),
        'signal_line': signal_line.tolist(),
        'histogram': histogram.tolist()
    }

def calculate_macd_pure_python(data, fast_period=12, slow_period=26, signal_period=9):
    """
    Pure Python implementation without pandas/numpy dependencies.
    
    Args:
        data: List of closing prices
        fast_period: Fast EMA period (default 12)
        slow_period: Slow EMA period (default 26)
        signal_period: Signal EMA period (default 9)
    
    Returns:
        dict: MACD line, signal line, and histogram
    """
    if not data or len(data) < slow_period:
        return None
    
    def calculate_ema(values, period):
        ema = [None] * len(values)
        multiplier = 2 / (period + 1)
        
        # Calculate initial SMA
        sma = sum(values[:period]) / period
        ema[period - 1] = sma
        
        # Calculate EMA
        for i in range(period, len(values)):
            ema[i] = (values[i] - ema[i - 1]) * multiplier + ema[i - 1]
        
        return ema
    
    # Calculate EMAs
    ema_fast = calculate_ema(data, fast_period)
    ema_slow = calculate_ema(data, slow_period)
    
    # Calculate MACD line
    macd_line = [None] * len(data)
    for i in range(slow_period, len(data)):
        if ema_fast[i] is not None and ema_slow[i] is not None:
            macd_line[i] = ema_fast[i] - ema_slow[i]
    
    # Calculate signal line
    signal_line = [None] * len(data)
    # Use MACD line values for signal EMA
    macd_values = [x for x in macd_line if x is not None]
    if macd_values:
        signal_ema = calculate_ema(macd_values, signal_period)
        # Map back to original indices
        sig_idx = 0
        for i in range(len(data)):
            if macd_line[i] is not None:
                if sig_idx < len(signal_ema):
                    signal_line[i] = signal_ema[sig_idx]
                    sig_idx += 1
    
    # Calculate histogram
    histogram = [None] * len(data)
    for i in range(len(data)):
        if macd_line[i] is not None and signal_line[i] is not None:
            histogram[i] = macd_line[i] - signal_line[i]
    
    return {
        'macd_line': macd_line,
        'signal_line': signal_line,
        'histogram': histogram
    }

Troubleshooting

MACD returns all None values

Cause: The input data array is too short (less than 26 periods) for the slow EMA calculation to initialize properly.

Fix: Ensure your data array contains at least 26 data points and that the slow_period parameter is less than the data length.

numpy or pandas not found

Cause: Dependencies are not installed in the environment.

Fix: Install numpy and pandas using 'pip install numpy pandas' or use the pure Python implementation.

Histogram values are consistently positive or negative

Cause: The price data shows a strong one-directional trend without divergence, which is expected behavior in strong bull or bear markets.

Fix: This is not an error; it indicates strong momentum. Look for crossovers when the trend weakens or reverses.

Frequently Asked Questions

What is the standard MACD configuration in Python?

The standard MACD configuration uses a 12-period fast EMA, 26-period slow EMA, and 9-period signal line, implemented using pandas ewm for efficient calculation.

Why does MACD return None values at the beginning in Python?

MACD returns None for the first 25 data points because it needs at least 26 periods to calculate the initial EMA values before generating meaningful signals.

What does a positive histogram indicate in Python MACD?

A positive histogram indicates the MACD line is above the signal line, suggesting bullish momentum and potential buying opportunity.

Can MACD be used for cryptocurrency trading in Python?

Yes, the MACD indicator is widely used in cryptocurrency trading to identify momentum shifts and potential entry/exit points across digital assets.
