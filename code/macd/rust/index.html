use std::collections::VecDeque;

#[derive(Debug, Clone)]
pub struct MACDResult {
    pub macd_line: Vec<Option<f64>>,
    pub signal_line: Vec<Option<f64>>,
    pub histogram: Vec<Option<f64>>,
}

fn calculate_ema(data: &[f64], period: usize) -> Vec<Option<f64>> {
    if data.len() < period {
        return vec![None; data.len()];
    }

    let mut ema = vec![None; data.len()];
    let multiplier = 2.0 / (period as f64 + 1.0);

    // Calculate initial SMA
    let mut sum = 0.0;
    for i in 0..period {
        sum += data[i];
    }
    ema[period - 1] = Some(sum / period as f64);

    // Calculate EMA
    for i in period..data.len() {
        if let Some(prev_ema) = ema[i - 1] {
            ema[i] = Some((data[i] - prev_ema) * multiplier + prev_ema);
        }
    }

    ema
}

pub fn calculate_macd(
    data: &[f64],
    fast_period: usize,
    slow_period: usize,
    signal_period: usize,
) -> Option<MACDResult> {
    if data.is_empty() || data.len() < slow_period {
        return None;
    }

    let ema_fast = calculate_ema(data, fast_period);
    let ema_slow = calculate_ema(data, slow_period);

    let mut macd_line = vec![None; data.len()];
    for i in 0..data.len() {
        if let (Some(fast), Some(slow)) = (ema_fast[i], ema_slow[i]) {
            macd_line[i] = Some(fast - slow);
        }
    }

    // Collect MACD line values for signal EMA
    let mut macd_values: Vec<f64> = Vec::new();
    for val in &macd_line {
        if let Some(v) = val {
            macd_values.push(*v);
        }
    }

    let mut signal_line = vec![None; data.len()];
    if !macd_values.is_empty() {
        let signal_ema = calculate_ema(&macd_values, signal_period);
        let mut sig_idx = 0;
        for i in 0..data.len() {
            if macd_line[i].is_some() {
                if sig_idx < signal_ema.len() {
                    signal_line[i] = signal_ema[sig_idx];
                    sig_idx += 1;
                }
            }
        }
    }

    let mut histogram = vec![None; data.len()];
    for i in 0..data.len() {
        if let (Some(macd), Some(signal)) = (macd_line[i], signal_line[i]) {
            histogram[i] = Some(macd - signal);
        }
    }

    Some(MACDResult {
        macd_line,
        signal_line,
        histogram,
    })
}

Key Lines Explained

fn calculate_ema(data: &[f64], period: usize) -> Vec<Option<f64>>

The EMA calculation function returns a vector of Option values, using None for positions where the EMA cannot be calculated due to insufficient data.

let multiplier = 2.0 / (period as f64 + 1.0)

The EMA multiplier is calculated as 2/(period+1), which weights recent prices more heavily than older ones.

ema[i] = Some((data[i] - prev_ema) * multiplier + prev_ema)

The recursive EMA formula applies the multiplier to the difference between current price and previous EMA, creating a smooth average.

let ema_fast = calculate_ema(data, fast_period);

Two separate EMA calculations are performed on the price data to generate the fast and slow EMAs.

macd_line[i] = Some(fast - slow)

The MACD line is the difference between the fast and slow EMAs, representing the momentum and trend direction.

Expert Notes

Why This Approach?

This Rust implementation uses Option types and vector operations because it provides memory safety, zero-cost abstractions, and optimal performance for high-frequency trading applications.

Alternatives

Using ndarray for vectorized operations

✓ More efficient for large datasets, supports SIMD operations

✗ Adds external dependency, more complex API

Using VecDeque for sliding window operations

✓ Memory efficient for streaming data, no need to store all values

✗ More complex logic for EMA calculation

Troubleshooting

MACD returns None

Cause: The input data slice is empty or too short (less than 26 periods) for the slow EMA calculation to initialize properly.

Fix: Ensure your data contains at least 26 data points and that the slow_period parameter is less than the data length.

Cannot borrow data as mutable

Cause: Rust's ownership and borrowing rules are being violated when accessing or modifying the data.

Fix: Use references (&) or implement Clone/ Copy traits for the data structure to avoid ownership issues.

Histogram values are consistently positive or negative

Cause: The price data shows a strong one-directional trend without divergence, which is expected behavior in strong bull or bear markets.

Fix: This is not an error; it indicates strong momentum. Look for crossovers when the trend weakens or reverses.

Frequently Asked Questions

What is the standard MACD configuration in Rust?

The standard MACD configuration uses a 12-period fast EMA, 26-period slow EMA, and 9-period signal line, implemented using Option types for safe null handling.

Why does MACD return None values at the beginning in Rust?

MACD returns None for the first 25 data points because it needs at least 26 periods to calculate the initial EMA values before generating meaningful signals.

What does a positive histogram indicate in Rust MACD?

A positive histogram indicates the MACD line is above the signal line, suggesting bullish momentum and potential buying opportunity.

Can MACD be used for cryptocurrency trading in Rust?

Yes, the MACD indicator is widely used in cryptocurrency trading to identify momentum shifts and potential entry/exit points across digital assets.
