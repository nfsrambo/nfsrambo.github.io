import java.util.ArrayList;
import java.util.LinkedList;
import java.util.List;

public class RSI {
    private final int period;
    private final LinkedList<Double> gains;
    private final LinkedList<Double> losses;
    private double avgGain;
    private double avgLoss;
    private boolean initialized;

    public RSI(int period) {
        this.period = period;
        this.gains = new LinkedList<>();
        this.losses = new LinkedList<>();
        this.avgGain = 0.0;
        this.avgLoss = 0.0;
        this.initialized = false;
    }

    public RSI() {
        this(14);
    }

    public List<Double> calculate(double[] prices) {
        if (prices.length < 2) {
            return new ArrayList<>();
        }

        List<Double> rsiValues = new ArrayList<>(prices.length);

        for (int i = 1; i < prices.length; i++) {
            double delta = prices[i] - prices[i - 1];

            if (delta > 0.0) {
                gains.add(delta);
                losses.add(0.0);
            } else {
                gains.add(0.0);
                losses.add(-delta);
            }

            if (gains.size() > period) {
                gains.removeFirst();
                losses.removeFirst();
            }

            if (gains.size() == period) {
                if (!initialized) {
                    avgGain = gains.stream().mapToDouble(Double::doubleValue).sum() / period;
                    avgLoss = losses.stream().mapToDouble(Double::doubleValue).sum() / period;
                    initialized = true;
                } else {
                    double lastGain = gains.getLast();
                    double lastLoss = losses.getLast();

                    avgGain = ((avgGain * (period - 1)) + lastGain) / period;
                    avgLoss = ((avgLoss * (period - 1)) + lastLoss) / period;
                }

                double rs = (avgLoss > 0.0) ? (avgGain / avgLoss) : Double.POSITIVE_INFINITY;
                double rsi = Double.isInfinite(rs) ? 100.0 : 100.0 - (100.0 / (1.0 + rs));

                rsiValues.add(rsi);
            }
        }

        return rsiValues;
    }

    public void reset() {
        gains.clear();
        losses.clear();
        avgGain = 0.0;
        avgLoss = 0.0;
        initialized = false;
    }
}

Frequently Asked Questions

Is Java suitable for high-frequency trading?

Java is suitable for mid-frequency trading (< 1ms). HFT firms using Java often use off-heap memory, GC tuning, and JVM warm-up to achieve low-latency. For microsecond-level HFT, C++ or Rust is preferred.

What's the performance difference between Java and Python for RSI?

Java's JIT compilation makes it approximately 4.3x faster than Python for RSI calculations. The performance gap widens for larger datasets as JIT optimizations kick in.

Should I use Java 8 or newer for trading systems?

Use Java 17+ for production. It offers significant performance improvements, better GC (ZGC, Shenandoah), and modern features like Records, Pattern Matching, and enhanced Switch.
